# Secret/credential patterns - one regex per line
# Lines starting with # are comments

# OpenAI API keys
sk-[a-zA-Z0-9]{20,}

# AWS Access Key IDs
AKIA[0-9A-Z]{16}

# GitHub Personal Access Tokens
ghp_[a-zA-Z0-9]{36}

# GitHub OAuth tokens
gho_[a-zA-Z0-9]{36}

# npm tokens
npm_[a-zA-Z0-9]{36}

# Private keys
-----BEGIN.*PRIVATE KEY

# Slack tokens
xox[baprs]-[a-zA-Z0-9-]+

# Anthropic API keys
sk-ant-[a-zA-Z0-9-]+
